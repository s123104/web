grok_search.py 
import requests
import os

def search_with_grok(question, **params):
    """使用 Grok Live Search 搜尋資訊"""
    
    # API 設定
    api_key = os.getenv("XAI_API_KEY")
    url = "https://api.x.ai/experimental/chat/completions"
    
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }
    
    # 預設搜尋參數
    default_params = {
        "mode": "auto",
        "max_search_results": 10,
        "country": "TW",
        "return_citations": True,
        "safe_search": True
    }
    
    # 合併用戶參數
    search_params = {**default_params, **params}
    
    # 構建請求
    payload = {
        "model": "grok-3-latest",
        "messages": [{"role": "user", "content": question}],
        "search_parameters": search_params
    }
    
    try:
        response = requests.post(url, headers=headers, json=payload)
        response.raise_for_status()
        
        result = response.json()
        return {
            "answer": result["choices"][0]["message"]["content"],
            "sources": result.get("citations", [])
        }
    except Exception as e:
        return {"error": str(e)}

# 使用範例
if __name__ == "__main__":
    # 基本搜尋
    result = search_with_grok("今天台灣有什麼重要新聞？")
    print("回答:", result["answer"])
    print("來源:", result["sources"])
    
    # 自訂參數搜尋
    tech_result = search_with_grok(
        "最新的 AI 技術趨勢",
        max_search_results=15,
        mode="on"
    )
    print(tech_result["answer"])
grok-search.js 
async function searchWithGrok(question, params = {}) {
    const apiKey = process.env.XAI_API_KEY;
    const url = "https://api.x.ai/experimental/chat/completions";
    
    // 預設參數
    const defaultParams = {
        mode: "auto",
        max_search_results: 10,
        country: "TW",
        return_citations: true,
        safe_search: true
    };
    
    // 合併參數
    const searchParams = { ...defaultParams, ...params };
    
    const payload = {
        model: "grok-3-latest",
        messages: [{ role: "user", content: question }],
        search_parameters: searchParams
    };
    
    try {
        const response = await fetch(url, {
            method: "POST",
            headers: {
                "Authorization": `Bearer ${apiKey}`,
                "Content-Type": "application/json"
            },
            body: JSON.stringify(payload)
        });
        
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        const result = await response.json();
        return {
            answer: result.choices[0].message.content,
            sources: result.citations || []
        };
    } catch (error) {
        return { error: error.message };
    }
}

// 使用範例
async function main() {
    // 基本搜尋
    const result = await searchWithGrok("今天台灣股市表現如何？");
    console.log("回答:", result.answer);
    console.log("來源:", result.sources);
    
    // 搜尋特定社群討論
    const socialResult = await searchWithGrok(
        "#AI 最新討論",
        { 
            sources: [{ type: "x" }],
            max_search_results: 20 
        }
    );
    console.log(socialResult.answer);
}

main().catch(console.error);

常見問題

快速解答新手最常遇到的問題

Live Search 功能是免費的嗎？

Beta 期間（至 2025/6/5）完全免費使用。正式版可能會有使用限制或費用，請關注官方更新。

搜尋結果的準確性如何？

Grok 會從多個可信來源獲取資訊，但仍建議檢查返回的引用來源。搜尋結果的準確性取決於來源品質。

API 有使用限制嗎？

Beta 期間預設每分鐘 50 請求。如需更高限制可向 xAI 申請。建議合理使用避免被限流。

如何處理 API 錯誤？

常見錯誤包括金鑰無效、請求超限、服務暫停等。建議實作重試機制和錯誤日誌記錄。
