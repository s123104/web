Arduino防盜系統介紹

專題簡介 材料介紹 電路圖 流程圖 程式碼解說

專題簡介

這個專題的目標是製作一套防盜系統，運作原理就像一家店的警報系統：當「保全攝影機」（光敏電阻）偵測到異常光線（好比小偷現形）時，就立即啟動警報（蜂鳴器）並點亮警示燈（LED），提醒主人或嚇退賊人。

材料介紹

材料名稱 功能說明
光敏電阻 (LDR) 好比保全攝影機的鏡頭，用來偵測環境光線的變化
LED 燈 當警報觸發時亮起，像店門口的閃光燈提醒大家注意
蜂鳴器 觸發時發出警報聲音，就像店家大聲喊話嚇退賊人
電阻 調節LED的電流，避免元件因過電流而受損，就像給器件穿上安全衣
Arduino 主控板 防盜系統的大腦，負責接收訊號並下達命令
杜邦線和麵包板 用來串聯所有元件，像是連接各個部件的神經網絡

系統流程圖

這個流程圖詳述了防盜系統的運作邏輯：當光敏電阻偵測到環境變化（異常光線），系統就會啟動警報並點亮燈號，類似於店內保全系統自動啟動警告程序。

Arduino 程式碼解說

/* 定義光敏電阻接腳，接在A0，像保全攝影機接收環境訊號 */
int ldrPin = A0; // 用來讀取環境光線數值

/* 定義LED接腳，接在D2，當警報觸發時會亮起 */
int ledPin = 2; // 控制LED燈的開關

/* 定義蜂鳴器接腳，接在D3，觸發時發出警報聲 */
int buzzerPin = 3; // 控制蜂鳴器的開關

/* setup函式只會在程式開始時執行一次，就像啟動系統前做基本設定 */
void setup() {
  pinMode(ledPin, OUTPUT);    // 設定LED接腳為輸出，能控制燈的狀態
  pinMode(buzzerPin, OUTPUT); // 設定蜂鳴器接腳為輸出，能控制警報聲
  Serial.begin(9600);         // 初始化序列監控，讓你可以在電腦上看到光線數值的變化
}

/* loop函式會不斷重複執行，就像防盜系統持續監控環境 */
void loop() {
  int ldrValue = analogRead(ldrPin); // 讀取光敏電阻的數值，就像保全攝影機捕捉到的光線資訊
  
  Serial.println(ldrValue); // 輸出光線數值到序列監控，方便你調整警報靈敏度
  
  // 如果讀取的光線數值超過800，代表環境亮度異常（好比偵測到可疑人物出現）
  if (ldrValue > 800) {
    digitalWrite(ledPin, HIGH);    // 打開LED燈，發出警示光線
    digitalWrite(buzzerPin, HIGH); // 啟動蜂鳴器，發出警報聲
  } else {
    digitalWrite(ledPin, LOW);     // 關閉LED燈
    digitalWrite(buzzerPin, LOW);  // 關閉蜂鳴器
  }
  
  delay(200); // 暫停200毫秒，避免系統執行過快
}
